linux常用命令

2014-12-22 linux

操作命令

# 移动某文件/文件夹到 某目录
mv phpmyadmin ../blog/  
# 删除某文件/文件/全部文件
rm -rf <rootname>/*
# 更改文件夹权限为 www 权限
chown -R www:www blog/
# 下载文件
wget <href>     
# 解压文件
tar zxvf  文件.tar.gz

PHP操作命令

# 查看mysql命令
/etc/init.d/mysql
# 查看nginx命令
/etc/init.d/nginx
# 查看php命令
/etc/init.d/php-fpm
# 重启nginx服务
/etc/init.d/nginx restart

设置fish作默认shell

# 查看fish 路径
type fish
# 设置fish作默认shell
chsh -s /usr/bin/fish
